let rates = {}; // 페이지 로드 시 환율 자동으로 가져오기 async function fetchRates() { try { const response = await fetch('https://api.exchangerate-api.com/v4/latest/KRW'); const data = await response.json(); rates = { 'CNY': { rate: data.rates.CNY, name: '중국 위안(CNH)' }, 'USD': { rate: data.rates.USD, name: '미국 달러(USD)' }, 'JPY': { rate: data.rates.JPY, name: '일본 엔(JPY)' }, 'EUR': { rate: data.rates.EUR, name: '유로(EUR)' }, 'GBP': { rate: data.rates.GBP, name: '영국 파운드(GBP)' }, 'HKD': { rate: data.rates.HKD, name: '홍콩 달러(HKD)' }, 'SGD': { rate: data.rates.SGD, name: '싱가포르 달러(SGD)' }, 'THB': { rate: data.rates.THB, name: '태국 바트(THB)' } }; console.log('환율이 업데이트되었습니다:', rates); } catch (error) { console.error('환율 로드 실패:', error); rates = { 'CNY': { rate: 0.000218, name: '중국 위안(CNH)' }, 'USD': { rate: 0.000653, name: '미국 달러(USD)' }, 'JPY': { rate: 0.00657, name: '일본 엔(JPY)' }, 'EUR': { rate: 0.000715, name: '유로(EUR)' }, 'GBP': { rate: 0.000819, name: '영국 파운드(GBP)' }, 'HKD': { rate: 0.00514, name: '홍콩 달러(HKD)' }, 'SGD': { rate: 0.000533, name: '싱가포르 달러(SGD)' }, 'THB': { rate: 0.0224, name: '태국 바트(THB)' } }; } } function calculate() { const amount = parseFloat(document.getElementById('amount').value); const currency = document.getElementById('currency').value; const rate = rates[currency]; if (!rate || isNaN(amount) || amount < 0) { document.getElementById('resultKRW').textContent = '0'; document.getElementById('result').textContent = '0'; document.getElementById('currencyLabel').textContent = '환율'; return; } const result = amount * rate.rate; document.getElementById('resultKRW').textContent = amount.toLocaleString(); document.getElementById('result').textContent = result.toFixed(2); document.getElementById('currencyLabel').textContent = currency; document.getElementById('exchangeRate').textContent = `1 KRW = ${rate.rate.toFixed(6)} ${currency}`; document.getElementById('updateTime').textContent = new Date().toLocaleString('ko-KR'); } // 이벤트 리스너 document.getElementById('amount').addEventListener('input', calculate); document.getElementById('currency').addEventListener('change', calculate); // 페이지 로드 시 환율 가져오기 window.addEventListener('load', function() { fetchRates(); calculate(); }); // 5분마다 자동 새로고침 setInterval(fetchRates, 5 * 60 * 1000);
